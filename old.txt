<!DOCTYPE html>
<html lang="en">

<head>
  <%- include("./partials/head") %>
  <title>Blog</title>
</head>

<body>
  <%- include("./partials/nav") %>

  <div class="container mt-4">
    <h1><%= blog.title %></h1>
    <img src="<%= blog.coverImageURL %>" width="700px">
    <pre><%= blog.body %></pre>
  </div>

  <!-- this is for user profile -->

  <div class="container mt-4">
    <img src="<%= blog.createdBy.profileImageURL %>" width="60px" />
    <%= blog.createdBy.fullname %>
  </div>

  <div class="container mt-3">
    <h1>Comments (<%= comments.length %>) </h1>

    <% if (locals.user) { %>
    <form action="/blog/comment/<%= blog._id %>" method="post">

      <div class="mb-3">
        <input type="text" class="content" name="content" placeholder="Enter your Comment" />
        <button class="btn btn-sm btn-primary" type="submit">Add</button>
      </div>
    </form>
    <% } %>

    <div class="mt-3">
    <% comments.forEach(comment => { %>
     <div>
    <img src="<%= comment.createdBy.profileImageURL %>" width="60px" />
    <%= comment.createdBy.fullname %>
    <pre><%= comment.content %></pre> 
     </div>
    <% }) %>
   </div>
  </div>

  <%- include("./partials/script") %>
</body>

</html>
